Array Implementation

C++



#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    vector<int> arr;

    // Add elements
    arr.push_back(10);
    arr.push_back(20);
    arr.push_back(30);

    // Remove element 20
    arr.erase(remove(arr.begin(), arr.end(), 20), arr.end());

    // Search element 30
    auto it = find(arr.begin(), arr.end(), 30);
    if (it != arr.end()) {
        cout << "Found 30 at index " << distance(arr.begin(), it) << endl;
    } else {
        cout << "30 not found" << endl;
    }

    // Print array
    for (int val : arr) cout << val << " ";
    cout << endl;

    return 0;
}

Stack Implementation

C++



#include <iostream>
#include <stack>
using namespace std;

int main() {
    stack<int> stk;

    // Push elements
    stk.push(10);
    stk.push(20);
    stk.push(30);

    // Peek
    if (!stk.empty()) {
        cout << "Top element: " << stk.top() << endl;
    }

    // Pop
    while (!stk.empty()) {
        cout << "Popped: " << stk.top() << endl;
        stk.pop();
    }

    return 0;
}

Python


stack = []

def push(stack, item):
    stack.append(item)

def pop(stack):
    return stack.pop() if stack else None

def peek(stack):
    return stack[-1] if stack else None

def is_empty(stack):
    return len(stack) == 0

precedence = {'+':1, '-':1, '*':2, '/':2, '^':3}

def infix_to_postfix(expression):
    result = []
    stack = []
    for char in expression:
        if char.isalnum():
            result.append(char)
        elif char == '(':
            push(stack, char)
        elif char == ')':
            while not is_empty(stack) and peek(stack) != '(':
                result.append(pop(stack))
            pop(stack)  # Remove '('
        else:  # operator
            while (not is_empty(stack) and peek(stack) != '(' and
                   precedence.get(char, 0) <= precedence.get(peek(stack), 0)):
                result.append(pop(stack))
            push(stack, char)
    while not is_empty(stack):
        result.append(pop(stack))
    return ''.join(result)

def infix_to_prefix(expression):
    
    expression = expression[::-1]
    
    expression = ''.join(['(' if c == ')' else ')' if c == '(' else c for c in expression])
    
    postfix = infix_to_postfix(expression)
    
    prefix = postfix[::-1]
    return prefix

expr = "A*(B+C)/D"
print("Infix expression:", expr)
print("Postfix expression:", infix_to_postfix(expr))  
print("Prefix expression:", infix_to_prefix(expr))

Queue Implementation

C++



#include <iostream>
#include <queue>
using namespace std;

int main() {
    queue<int> q;

    // Enqueue
    q.push(10);
    q.push(20);
    q.push(30);

    // Peek front
    if (!q.empty()) {
        cout << "Front element: " << q.front() << endl;
    }

    // Dequeue
    while (!q.empty()) {
        cout << "Dequeued: " << q.front() << endl;
        q.pop();
    }

    return 0;
}

Binary Tree Traversal Implementation

C++



#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* left;
    Node* right;
    Node(int val) : data(val), left(nullptr), right(nullptr) {}
};

void preorder(Node* root) {
    if (!root) return;
    cout << root->data << " ";
    preorder(root->left);
    preorder(root->right);
}

void inorder(Node* root) {
    if (!root) return;
    inorder(root->left);
    cout << root->data << " ";
    inorder(root->right);
}

void postorder(Node* root) {
    if (!root) return;
    postorder(root->left);
    postorder(root->right);
    cout << root->data << " ";
}

int main() {
    // Create tree
    Node* root = new Node(1);
    root->left = new Node(2);
    root->right = new Node(3);
    root->left->left = new Node(4);
    root->left->right = new Node(5);
    root->right->left = new Node(6);
    root->right->right = new Node(7);

    cout << "Preorder: ";
    preorder(root);
    cout << "\nInorder: ";
    inorder(root);
    cout << "\nPostorder: ";
    postorder(root);
    cout << endl;

    return 0;
}

10 Algorithms

  • Binary Search: Efficiently searches a sorted array by repeatedly dividing the search interval in half.
  • Bubble Sort: Simple comparison-based sorting algorithm.
  • Selection Sort: Selects the minimum element and swaps with the beginning.
  • Insertion Sort: Builds the final sorted array one item at a time.
  • Merge Sort: Divide and conquer sorting algorithm.
  • Quick Sort: Efficient divide and conquer sorting algorithm.
  • Dijkstra's Algorithm: Finds the shortest path in a graph.
  • Depth First Search (DFS): Graph traversal algorithm.
  • Breadth First Search (BFS): Graph traversal algorithm.
  • Floyd-Warshall Algorithm: Finds shortest paths between all pairs of nodes in a graph.

10 Pseudocode Examples

  1. Binary Search: 
      
1. Set low = 0, high = length(array) - 1  
2. While low ≤ high:  
   3.   mid = (low + high) / 2  
   4.   If array[mid] == target, return mid  
   5.   Else if array[mid] < target, set low = mid + 1  
   6.   Else, set high = mid - 1  
7. Return -1 (not found)
  2. Bubble Sort: 
    1. For i from 0 to n-1:  
2.   For j from 0 to n-i-2:  
3.     If array[j] > array[j+1], swap them
  3. Selection Sort: 
    1. For i from 0 to n-1:  
2.   minIndex = i  
3.   For j from i+1 to n-1:  
4.     If array[j] < array[minIndex], minIndex = j  
5.   Swap array[i] and array[minIndex]
  4. Insertion Sort: 
    1. For i from 1 to n-1:  
2.   key = array[i]  
3.   j = i - 1  
4.   While j ≥ 0 and array[j] > key:  
5.     array[j+1] = array[j]  
6.     j = j - 1  
7.   array[j+1] = key
  5. Merge Sort: 
    1. If length(array) ≤ 1, return array  
2. Split array into two halves  
3. Sort left half recursively  
4. Sort right half recursively  
5. Merge the two sorted halves
  6. Quick Sort: 
    1. If low < high:  
2.   pi = partition(array, low, high)  
3.   QuickSort(array, low, pi - 1)  
4.   QuickSort(array, pi + 1, high)
  7. Dijkstra's Algorithm: 
    1. Initialize distances from source to all vertices as infinity  
2. Set distance to source = 0  
3. Use priority queue to select vertex with minimum distance  
4. Update distances of adjacent vertices  
5. Repeat until all vertices processed
  8. Depth First Search (DFS): 
    1. Start at a node, mark as visited  
2. For each adjacent node not visited, recursively DFS
  9. Breadth First Search (BFS): 
    1. Start at a node, enqueue it  
2. While queue not empty:  
3.   Dequeue node, visit it  
4.   Enqueue all adjacent unvisited nodes
  10. Floyd-Warshall Algorithm: 
    1. Initialize distance matrix  
2. For each vertex k:  
3.   For each vertex i:  
4.     For each vertex j:  
5.       If dist[i][k] + dist[k][j] < dist[i][j]:  
6.         dist[i][j] = dist[i][k] + dist[k][j]
