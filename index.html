Pseudocode Examples

Bubble Sort


FOR i = 0 to n-1
  FOR j = 0 to n-i-1
    IF array[j] > array[j+1]
      SWAP array[j] and array[j+1]
    END IF
  END FOR
END FOR

Binary Search


SET low = 0
SET high = n - 1

WHILE low <= high
  SET mid = (low + high) / 2
  IF array[mid] == target
    RETURN mid
  ELSE IF array[mid] < target
    SET low = mid + 1
  ELSE
    SET high = mid - 1
  END IF
END WHILE
RETURN -1  // Not found

Array (Python)



arr = []

def insert_element(arr, element):
    arr.append(element)

def delete_element(arr, element):
    if element in arr:
        arr.remove(element)

def traverse(arr):
    for item in arr:
        print(item, end=' ')
    print()

insert_element(arr, 10)
insert_element(arr, 20)
print("Array after insertions:")
traverse(arr)  # Output: 10 20
delete_element(arr, 10)
print("Array after deletion:")
traverse(arr)  # Output: 20

Stack with Infix to Postfix and Prefix Conversion (Python)



stack = []

def push(stack, item):
    stack.append(item)

def pop(stack):
    return stack.pop() if stack else None

def peek(stack):
    return stack[-1] if stack else None

def is_empty(stack):
    return len(stack) == 0

precedence = {'+':1, '-':1, '*':2, '/':2, '^':3}

def infix_to_postfix(expression):
    result = []
    stack = []
    for char in expression:
        if char.isalnum():
            result.append(char)
        elif char == '(':
            push(stack, char)
        elif char == ')':
            while not is_empty(stack) and peek(stack) != '(':
                result.append(pop(stack))
            pop(stack)  # Remove '('
        else:  # operator
            while (not is_empty(stack) and peek(stack) != '(' and
                   precedence.get(char, 0) <= precedence.get(peek(stack), 0)):
                result.append(pop(stack))
            push(stack, char)
    while not is_empty(stack):
        result.append(pop(stack))
    return ''.join(result)

def infix_to_prefix(expression):
    
    expression = expression[::-1]
    
    expression = ''.join(['(' if c == ')' else ')' if c == '(' else c for c in expression])
    
    postfix = infix_to_postfix(expression)
    
    prefix = postfix[::-1]
    return prefix

expr = "A*(B+C)/D"
print("Infix expression:", expr)
print("Postfix expression:", infix_to_postfix(expr))  # Output: ABC+*D/
print("Prefix expression:", infix_to_prefix(expr))    # Output: /*A+BCD

Queue (Python)



queue = []

def enqueue(queue, item):
    queue.append(item)

def dequeue(queue):
    return queue.pop(0) if queue else None

enqueue(queue, 1)
enqueue(queue, 2)
print("Dequeued item:", dequeue(queue))  # Outputs: 1
print("Queue after dequeue:", queue)     # Outputs: [2]

Tree Traversals (Python)


        
class Node:
    def __init__(self, key):
        self.left = None
        self.right = None
        self.val = key

def inorder(root):
    if root:
        inorder(root.left)
        print(root.val, end=' ')
        inorder(root.right)

def preorder(root):
    if root:
        print(root.val, end=' ')
        preorder(root.left)
        preorder(root.right)

def postorder(root):
    if root:
        postorder(root.left)
        postorder(root.right)
        print(root.val, end=' ')

root = Node(1)
root.left = Node(2)
root.right = Node(3)

print("Inorder traversal:")
inorder(root)  # Output: 2 1 3
print()

print("Preorder traversal:")
preorder(root)  # Output: 1 2 3
print()

print("Postorder traversal:")
postorder(root)  # Output: 2 3 1
print()
