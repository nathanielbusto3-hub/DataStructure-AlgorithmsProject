Pseudocode Examples

1. Bubble Sort

FOR i = 0 to n-1
  FOR j = 0 to n-i-1
    IF array[j] > array[j+1]
      SWAP array[j] and array[j+1]
    END IF
  END FOR
END FOR

2. Selection Sort

FOR i = 0 to n-1
  minIndex = i
  FOR j = i+1 to n-1
    IF array[j] < array[minIndex]
      minIndex = j
    END IF
  END FOR
  SWAP array[i] and array[minIndex]
END FOR

3. Insertion Sort

FOR i = 1 to n-1
  key = array[i]
  j = i - 1
  WHILE j >= 0 AND array[j] > key
    array[j+1] = array[j]
    j = j - 1
  END WHILE
  array[j+1] = key
END FOR

4. Linear Search

FOR i = 0 to n-1
  IF array[i] == target
    RETURN i
  END IF
END FOR
RETURN -1

5. Binary Search

low = 0
high = n - 1
WHILE low <= high
  mid = (low + high) / 2
  IF array[mid] == target
    RETURN mid
  ELSE IF array[mid] < target
    low = mid + 1
  ELSE
    high = mid - 1
  END IF
END WHILE
RETURN -1

6. Merge Sort

FUNCTION mergeSort(array)
  IF length(array) > 1
    mid = length(array) / 2
    leftHalf = array[0..mid]
    rightHalf = array[mid..end]
    mergeSort(leftHalf)
    mergeSort(rightHalf)
    merge(leftHalf, rightHalf into array)
  END IF
END FUNCTION

7. Quick Sort

FUNCTION quicksort(array, low, high)
  IF low < high
    pivotIndex = partition(array, low, high)
    quicksort(array, low, pivotIndex - 1)
    quicksort(array, pivotIndex + 1, high)
  END IF
END FUNCTION

8. Fibonacci

FUNCTION fibonacci(n)
  IF n <= 1
    RETURN n
  END IF
  RETURN fibonacci(n-1) + fibonacci(n-2)
END FUNCTION

9. Factorial

FUNCTION factorial(n)
  IF n == 0
    RETURN 1
  END IF
  RETURN n * factorial(n - 1)
END FUNCTION

10. Euclidean Algorithm (GCD)

FUNCTION gcd(a, b)
  WHILE b != 0
    temp = b
    b = a MOD b
    a = temp
  END WHILE
  RETURN a
END FUNCTION

Array (Python)



arr = []

def insert_element(arr, element):
    arr.append(element)

def delete_element(arr, element):
    if element in arr:
        arr.remove(element)

def traverse(arr):
    for item in arr:
        print(item, end=' ')
    print()

insert_element(arr, 10)
insert_element(arr, 20)
print("Array after insertions:")
traverse(arr)  # Output: 10 20
delete_element(arr, 10)
print("Array after deletion:")
traverse(arr)  # Output: 20

Stack with Infix to Postfix and Prefix Conversion (Python)



stack = []

def push(stack, item):
    stack.append(item)

def pop(stack):
    return stack.pop() if stack else None

def peek(stack):
    return stack[-1] if stack else None

def is_empty(stack):
    return len(stack) == 0

precedence = {'+':1, '-':1, '*':2, '/':2, '^':3}

def infix_to_postfix(expression):
    result = []
    stack = []
    for char in expression:
        if char.isalnum():
            result.append(char)
        elif char == '(':
            push(stack, char)
        elif char == ')':
            while not is_empty(stack) and peek(stack) != '(':
                result.append(pop(stack))
            pop(stack)  # Remove '('
        else:  # operator
            while (not is_empty(stack) and peek(stack) != '(' and
                   precedence.get(char, 0) <= precedence.get(peek(stack), 0)):
                result.append(pop(stack))
            push(stack, char)
    while not is_empty(stack):
        result.append(pop(stack))
    return ''.join(result)

def infix_to_prefix(expression):
    
    expression = expression[::-1]
    
    expression = ''.join(['(' if c == ')' else ')' if c == '(' else c for c in expression])
    
    postfix = infix_to_postfix(expression)
    
    prefix = postfix[::-1]
    return prefix

# Example
expr = "a+b*(c^d-e)^(f+g*h)-i"
print("Postfix:", infix_to_postfix(expr))
print("Prefix:", infix_to_prefix(expr))

Queue (Python)



from collections import deque

queue = deque()

def enqueue(q, item):
    q.append(item)

def dequeue(q):
    return q.popleft() if q else None

def is_empty(q):
    return len(q) == 0

enqueue(queue, 1)
enqueue(queue, 2)
print("Queue:")
while not is_empty(queue):
    print(dequeue(queue), end=' ')
print()

Tree Traversals (Python)



class Node:
    def __init__(self, key):
        self.left = None
        self.right = None
        self.val = key

def inorder(root):
    if root:
        inorder(root.left)
        print(root.val, end=' ')
        inorder(root.right)

def preorder(root):
    if root:
        print(root.val, end=' ')
        preorder(root.left)
        preorder(root.right)

def postorder(root):
    if root:
        postorder(root.left)
        postorder(root.right)
        print(root.val, end=' ')

# Example Tree
root = Node(1)
root.left = Node(2)
root.right = Node(3)

print("Inorder:")
inorder(root)
print("\nPreorder:")
preorder(root)
print("\nPostorder:")
postorder(root)
